/dev/mem_hint
A kernel control plane
for AI memory systems.

AI runtimes already know when they are entering prefill, decode, agentic execution, idle gaps, forward propagation, and gradient-heavy backward phases. Memory hardware does not. It sees transactions, queue pressure, retries, and late symptoms. /dev/mem_hint is the missing control-plane contract: one compact hint structure that lets software express workload intent while leaving timing, electrical, and equalization enforcement to a privileged policy engine and a non-bypassable safety layer.

This reference implementation demonstrates how explicit runtime hints, PMU classification, and operator policy controls can converge into one kernel-mediated signaling path. The driver is intentionally honest about what is real and what remains illustrative: the interface contract is concrete, the hardware conduits are reference embodiments, and the safety logic models the shape of what must eventually live in hardware.

§01 · The problem

AI software has intent. Memory hardware sees transactions.

Modern AI runtimes are phase-rich. A serving stack knows when it is swallowing an entire prompt, when it has entered a latency-sensitive decode loop, when an agent is oscillating between planning and tool I/O, and when a training job has moved from bandwidth-saturating forward propagation into write-heavy backward propagation. The memory system still sees all of this as command streams and queue pressure. It reacts after the fact.

That is the central mismatch behind /dev/mem_hint. The runtime has semantic knowledge. The controller firmware and PHY have knobs. The kernel is the only layer positioned to validate privileged requests, arbitrate multiple hint sources, and broker them into a channel the hardware side can consume. The patent’s core insight is that a predictive control plane is more valuable than a purely reactive one: the right time to adapt memory policy is when software knows a phase boundary is coming, not after latency histograms have already degraded.

Core insight

Predictive signaling wins because phase semantics arrive before the hardware symptoms. A prefill launch is known at queue dispatch time; a decode bottleneck is only visible after the controller has already accumulated misses, retries, and queueing delay.

Bridge

/dev/mem_hint bridges this gap by giving user space a stable, 8-byte intent format and giving the kernel a privileged control plane that can map that intent to policy without exposing raw DRAM internals to the runtime.

§02 · The 64-bit workload hint

One register write. Full semantic intent.

C
struct mem_workload_hint {
    uint8_t  phase_id;            /* [7:0]   phase semantic */
    uint16_t latency_target_ns;   /* [23:8]  latency objective */
    uint16_t bw_target_gbps;      /* [39:24] bandwidth objective */
    uint8_t  security_level;      /* [47:40] TEE / hardening intent */
    uint8_t  priority;            /* [55:48] 0..7 policy aggressiveness */
    uint8_t  reserved[2];         /* [63:56] reserved for future rail hints */
} __attribute__((packed));

/* Encoded 64-bit layout:
 * [63:56] reserved
 * [55:48] priority
 * [47:40] security_level
 * [39:24] bw_target_gbps
 * [23:8]  latency_target_ns
 * [7:0]   phase_id
 */
Priority rail

Priority does not mean “ignore safety.” It means “how far from nominal should policy move toward the phase-optimal setting before the safety limiter clamps it back.” Priority 0 means stay close to nominal JEDEC behavior. Priority 7 means fully chase the phase table, then let hardware-enforced limits decide the final answer.

C
static ssize_t mem_hint_write(struct file *file,
                                  const char __user *buf,
                                  size_t len, loff_t *ppos)
{
    struct mem_workload_hint hint;
    int ret;

    if (!capable(CAP_SYS_ADMIN))
        return -EPERM;
    if (len < sizeof(hint))
        return -EINVAL;
    if (copy_from_user(&hint, buf, sizeof(hint)))
        return -EFAULT;

    if (hint.phase_id < PHASE_PREFILL || hint.phase_id > PHASE_BACKWARD)
        return -EINVAL;
    if (hint.priority > 7)
        hint.priority = 7;
    if (hint.security_level > 0)
        pr_warn_ratelimited("TEE validation is illustrative");

    ret = mem_hint_apply(&hint);
    if (ret)
        return ret;

    atomic_set(&current_phase_id, hint.phase_id);
    sysfs_notify(NULL, "status", "current_phase");
    return sizeof(hint);
}
C
static u64 encode_hint(const struct mem_workload_hint *h)
{
    u64 val = 0;
    val |= (u64)h->phase_id;
    val |= ((u64)h->latency_target_ns & 0xffffULL) << 8;
    val |= ((u64)h->bw_target_gbps & 0xffffULL) << 24;
    val |= ((u64)h->security_level & 0xffULL) << 40;
    val |= ((u64)h->priority & 0xffULL) << 48;
    return val;
}

int mem_hint_apply(const struct mem_workload_hint *h)
{
    u64 val = encode_hint(h);
    switch (active_channel) {
    case CH_MSR:
        if (mem_hint_should_touch_hw())
            wrmsrl(MEM_HINT_MSR, val);
        break;
    case CH_MMIO:
        if (mc_mmio_base)
            iowrite64(val, mc_mmio_base + MEM_HINT_MMIO_OFFSET);
        break;
    case CH_CXL_DVSEC:
        pr_debug("illustrative CXL DVSEC path: 0x%llx", val);
        break;
    }
    return 0;
}
Illustrative MSR note

The MSR number in the reference driver is deliberately illustrative. The point of the implementation is the interface contract and privilege boundary, not a claim that any current x86 or ARM platform already exposes this exact register.

C
typedef struct mem_workload_hint {
    uint8_t  phase_id;
    uint16_t latency_target_ns;
    uint16_t bw_target_gbps;
    uint8_t  security_level;
    uint8_t  priority;
    uint8_t  reserved[2];
} mem_hint_t;

static inline int mem_hint_send(int fd, mem_hint_t hint)
{
    return write(fd, &hint, sizeof(hint)) == sizeof(hint) ? 0 : -1;
}

mem_hint_t prefill = { .phase_id = 0x01, .latency_target_ns = 200,
                       .bw_target_gbps = 400, .priority = 7 };
mem_hint_t decode  = { .phase_id = 0x02, .latency_target_ns = 90,
                       .bw_target_gbps = 150, .priority = 7 };

mem_hint_send(fd, prefill);
mem_hint_send(fd, decode);
Design principle

The runtime never needs to know tRCD, DFE Tap1, or V_swing. It only declares intent. Policy selection, hardware dispatch, and safety enforcement stay below the ABI boundary.

§05 · Automatic path — PMU classification

Works on unmodified runtimes. Zero application changes.

Not every deployment can patch vLLM, TensorRT-LLM, or a training stack immediately. The PMU path lets the kernel infer workload phase from memory telemetry. It is less exact than explicit hints, but it preserves the same policy machinery and the same safety model.

Event Source Interpretation
UNC_M_CAS_COUNT.WR IMC uncore High write bandwidth points toward Prefill
UNC_M_CAS_COUNT.RD IMC uncore High read with low write points toward Decode
MEM_LOAD_RETIRED.L3_MISS Core PMU Decode and Agentic miss pressure discriminator
OFFCORE_REQUESTS Core PMU Bandwidth variance signal for Agentic mode
UNC_M_CMD_RATE IMC uncore <1000/s implies Idle
C
struct pmu_sample {
    u32 write_bw_gbps;
    u32 read_bw_gbps;
    u32 llc_miss_rate;
    u32 bw_variance_pct;
    u32 dram_cmd_rate;
};

static u8 classify_phase(const struct pmu_sample *s, u8 prev)
{
    if (s->write_bw_gbps > 10 && s->write_bw_gbps > s->read_bw_gbps)
        return PHASE_PREFILL;
    if (s->read_bw_gbps > s->write_bw_gbps * 2 &&
        s->write_bw_gbps < 1 && s->llc_miss_rate > 5000)
        return PHASE_DECODE;
    if (s->bw_variance_pct > 50)
        return PHASE_AGENTIC;
    if (s->dram_cmd_rate < 1000)
        return PHASE_IDLE;
    return prev; /* hysteresis */
}

static irqreturn_t pmu_overflow_handler(...)
{
    struct pmu_sample s = collect_pmu_sample();
    u8 next = classify_phase(&s, atomic_read(&current_phase_id));
    /* synthesize mem_workload_hint and call mem_hint_apply() */
    return IRQ_HANDLED;
}

Closed loop

sysfs is the operator face of the closed-loop control system. ECC telemetry and latency feedback tell the safety layer when to back off; sysfs lets the operator tune how aggressively explicit hints or PMU inference attempt to move the system before the hardware clamp intervenes.

§08 · Safety — hardware root of trust

Software expresses intent. Hardware enforces the law.

A workload-aware memory interface without a hard safety boundary would be irresponsible. The moment software can ask for lower timing, more aggressive electrical drive, or region-specific behavior, the memory system becomes part of the attack surface. That includes stability attacks, thermal margin abuse, and any mechanism that intentionally reduces timing guard bands.

The right analogy is Rowhammer. Rowhammer mattered because the interface between software activity and physical memory behavior was tighter than people assumed. /dev/mem_hint takes that lesson seriously: software may propose a phase-appropriate mode, but hardware is the final authority. The limiter in this repository is a software model of that idea, not a substitute for the immutable ASIC logic described by the patent.

What the safety limiter prevents

It prevents unsafe requests such as tRCD=10 on a platform whose SPD floor is 14, over-driving a marginal channel with vswing=800mV, or allowing a high-security context to run with a decode profile that undercuts the system’s conservative nominal floor.

C
struct phy_config {
    u8  tRCD, tCL, tRP, tRAS;
    u16 vswing_mv;
    u8  dfe_tap[4];
    s8  ctle_gain_db;
    u8  refresh_mode;
};

struct jedec_limits {
    u8 min_tRCD, max_tRCD;
    u8 min_tCL,  max_tCL;
    u8 min_tRP,  max_tRP;
    u16 min_vswing_mv, max_vswing_mv;
};

static struct phy_config safety_clamp(struct phy_config p,
                                     struct jedec_limits l,
                                     struct ecc_telemetry e)
{
    p.tRCD = clamp_t(u8, p.tRCD, l.min_tRCD, l.max_tRCD);
    p.tCL  = clamp_t(u8, p.tCL,  l.min_tCL,  l.max_tCL);
    p.tRP  = clamp_t(u8, p.tRP,  l.min_tRP,  l.max_tRP);
    p.vswing_mv = clamp_t(u16, p.vswing_mv,
                            l.min_vswing_mv, l.max_vswing_mv);
    if (e.correctable_rate > 1000000ULL) {
        p.tRCD = min_t(u8, p.tRCD + 1, l.max_tRCD);
        p.tCL  = min_t(u8, p.tCL  + 1, l.max_tCL);
    }
    if (security_level > 0) {
        p.tRCD = max_t(u8, p.tRCD, 22);
        p.tCL  = max_t(u8, p.tCL,  22);
    }
    return p;
}

§10 · Closing

Memory needs an intent interface. Now it has one.

AI infrastructure already evolved past the idea that compute alone sets performance. Memory policy, fabric topology, cache residency, queue geometry, and latency tail behavior all shape end-to-end usefulness. Yet memory hardware is still usually driven as if phase intent were unknowable.

/dev/mem_hint is a concrete answer to that gap. It gives runtimes a minimal semantic vocabulary, gives the kernel a privileged place to validate and multiplex that intent, and gives hardware designers a clean boundary where policy and safety can meet.

This repository is a reference implementation, not a shipping hardware enablement package. But the contract is real, the software surfaces are buildable, and the architecture points toward a future where memory policy can finally become phase-aware instead of permanently reactive.
